50 // configure the system to get its clock from the PLL
51 // SYSDIV = 400/freq -1
52 // bus frequency is 400MHz/(SYSDIV+1)
53 void PLL_Init(INT32U freq){
54  // 0) configure the system to use RCC2 for advanced features
55  // such as 400 MHz PLL and non-integer System Clock Divisor
56  SYSCTL_RCC2_R |= SYSCTL_RCC2_USERCC2;
57  // 1) bypass PLL while initializing
58  SYSCTL_RCC2_R |= SYSCTL_RCC2_BYPASS2;
59  // 2) select the crystal value and oscillator source
60  SYSCTL_RCC_R &= ~SYSCTL_RCC_XTAL_M; // clear XTAL field
61  SYSCTL_RCC_R += SYSCTL_RCC_XTAL_16MHZ;// configure for 16 MHz crystal
62  SYSCTL_RCC2_R &= ~SYSCTL_RCC2_OSCSRC2_M;// clear oscillator source field
63  SYSCTL_RCC2_R += SYSCTL_RCC2_OSCSRC2_MO;// configure for main oscillator source
64  // 3) activate PLL by clearing PWRDN
65  SYSCTL_RCC2_R &= ~SYSCTL_RCC2_PWRDN2;
66  // 4) set the desired system divider and the system divider least significant bit
67  SYSCTL_RCC2_R |= SYSCTL_RCC2_DIV400; // use 400 MHz PLL
68  SYSCTL_RCC2_R = (SYSCTL_RCC2_R&~0x1FC00000) // clear system clock divider field
69  + (freq<<22); // configure for 80 MHz clock
70  // 5) wait for the PLL to lock by polling PLLLRIS
71  while((SYSCTL_RIS_R&SYSCTL_RIS_PLLLRIS)==0){};
72  // 6) enable use of PLL by clearing BYPASS
73  SYSCTL_RCC2_R &= ~SYSCTL_RCC2_BYPASS2;
74 }
