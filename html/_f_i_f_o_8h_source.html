32 #define AddIndexFifo(NAME,SIZE,TYPE,SUCCESS,FAIL) \
33 uint32_t volatile NAME ## PutI; \
34 uint32_t volatile NAME ## GetI; \
35 TYPE static NAME ## Fifo [SIZE]; \
36 void NAME ## Fifo_Init(void){ long sr; \
37  sr = StartCritical(); \
38  NAME ## PutI = NAME ## GetI = 0; \
39  EndCritical(sr); \
40 } \
41 int NAME ## Fifo_Put (TYPE data){ \
42  if(( NAME ## PutI - NAME ## GetI ) & ~(SIZE-1)){ \
43  return(FAIL); \
44  } \
45  NAME ## Fifo[ NAME ## PutI &(SIZE-1)] = data; \
46  NAME ## PutI ## ++; \
47  return(SUCCESS); \
48 } \

50  if( NAME ## PutI == NAME ## GetI ){ \
51  return(FAIL); \
52  } \

54  NAME ## GetI ## ++; \
55  return(SUCCESS); \
56 } \

58  return ((unsigned short)( NAME ## PutI - NAME ## GetI )); \
