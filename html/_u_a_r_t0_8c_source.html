63 // Initialize UART0
64 // Baud rate is 115200 bits/sec
65 void UART_Init(void){
66  SYSCTL_RCGCUART_R |= 0x01; // activate UART0
67  SYSCTL_RCGCGPIO_R |= 0x01; // activate port A
68  RxFifo_Init(); // initialize empty FIFOs
69  TxFifo_Init();
70  UART0_CTL_R &= ~UART_CTL_UARTEN; // disable UART
71 // UART0_IBRD_R = 27; // IBRD = int(50,000,000 / (16 * 115,200)) = int(27.1267)
72 // UART0_FBRD_R = 8; // FBRD = int(0.1267 * 64 + 0.5) = 8
73  UART0_IBRD_R = 43; // IBRD = int(80,000,000 / (16 * 115,200)) = int(43.4027)
74  UART0_FBRD_R = 26; // FBRD = int(0.4027 * 64 + 0.5) = 26
75  // 8 bit word length (no parity bits, one stop bit, FIFOs)
76  UART0_LCRH_R = (UART_LCRH_WLEN_8|UART_LCRH_FEN);
77  UART0_IFLS_R &= ~0x3F; // clear TX and RX interrupt FIFO level fields
78  // configure interrupt for TX FIFO <= 1/8 full
79  // configure interrupt for RX FIFO >= 1/8 full
80  UART0_IFLS_R += (UART_IFLS_TX1_8|UART_IFLS_RX1_8);
81  // enable TX and RX FIFO interrupts and RX time-out interrupt
82  UART0_IM_R |= (UART_IM_RXIM|UART_IM_TXIM|UART_IM_RTIM);
83  UART0_CTL_R |= 0x301; // enable UART
84  GPIO_PORTA_AFSEL_R |= 0x03; // enable alt funct on PA1-0
85  GPIO_PORTA_DEN_R |= 0x03; // enable digital I/O on PA1-0
86  // configure PA1-0 as UART
87  GPIO_PORTA_PCTL_R = (GPIO_PORTA_PCTL_R&0xFFFFFF00)+0x00000011;
88  GPIO_PORTA_AMSEL_R = 0; // disable analog functionality on PA
89  // UART0=priority 2
90  NVIC_PRI1_R = (NVIC_PRI1_R&0xFFFF00FF)|0x00004000; // bits 13-15
91  NVIC_EN0_R = NVIC_EN0_INT5; // enable interrupt 5 in NVIC
92  OS_InitSemaphore(&semaUART, 1);
93  //EnableInterrupts();
94 }
95 // copy from hardware RX FIFO to software RX FIFO
96 // stop when hardware RX FIFO is empty or software RX FIFO is full
97 void static copyHardwareToSoftware(void){
98  char letter;
99  while(((UART0_FR_R&UART_FR_RXFE) == 0) && (RxFifo_Size() < (FIFOSIZE - 1))){
100  letter = UART0_DR_R;
101  RxFifo_Put(letter);
102  }
103 }
104 // copy from software TX FIFO to hardware TX FIFO
105 // stop when software TX FIFO is empty or hardware TX FIFO is full
106 void static copySoftwareToHardware(void){
107  char letter;
108  while(((UART0_FR_R&UART_FR_TXFF) == 0) && (TxFifo_Size() > 0)){
109  TxFifo_Get(&letter);
110  UART0_DR_R = letter;
111  }
112 }
113 // input ASCII character from UART
114 // spin if RxFifo is empty
115 char UART_InChar(void){
116  char letter;
117  while(RxFifo_Get(&letter) == FIFOFAIL){};
118  return(letter);
119 }
120 // output ASCII character to UART
121 // spin if TxFifo is full
122 void UART_OutChar(char data){
123  OS_Wait(&semaUART);
124  while(TxFifo_Put(data) == FIFOFAIL){};
125  UART0_IM_R &= ~UART_IM_TXIM; // disable TX FIFO interrupt
126  copySoftwareToHardware();
127  UART0_IM_R |= UART_IM_TXIM; // enable TX FIFO interrupt
128  OS_Signal(&semaUART);
129 }
130 // at least one of three things has happened:
131 // hardware TX FIFO goes from 3 to 2 or less items
132 // hardware RX FIFO goes from 1 to 2 or more items
133 // UART receiver has timed out
134 void UART0_Handler(void){
135  if(UART0_RIS_R&UART_RIS_TXRIS){ // hardware TX FIFO <= 2 items
136  UART0_ICR_R = UART_ICR_TXIC; // acknowledge TX FIFO
137  // copy from software TX FIFO to hardware TX FIFO
138  copySoftwareToHardware();
139  if(TxFifo_Size() == 0){ // software TX FIFO is empty
140  UART0_IM_R &= ~UART_IM_TXIM; // disable TX FIFO interrupt
141  }
142  }
143  if(UART0_RIS_R&UART_RIS_RXRIS){ // hardware RX FIFO >= 2 items
144  UART0_ICR_R = UART_ICR_RXIC; // acknowledge RX FIFO
145  // copy from hardware RX FIFO to software RX FIFO
146  copyHardwareToSoftware();
147  }
148  if(UART0_RIS_R&UART_RIS_RTRIS){ // receiver timed out
149  UART0_ICR_R = UART_ICR_RTIC; // acknowledge receiver time out
150  // copy from hardware RX FIFO to software RX FIFO
151  copyHardwareToSoftware();
152  }
153 }

166 //------------UART_InUDec------------
167 // InUDec accepts ASCII input in unsigned decimal format
168 // and converts to a 32-bit unsigned number
169 // valid range is 0 to 4294967295 (2^32-1)
170 // Input: none
171 // Output: 32-bit unsigned number
172 // If you enter a number above 4294967295, it will return an incorrect value
173 // Backspace will remove last digit typed
174 uint32_t UART_InUDec(void){
175 uint32_t number=0, length=0;
176 char character;
177  character = UART_InChar();
178  while(character != CR){ // accepts until <enter> is typed
179 // The next line checks that the input is a digit, 0-9.
180 // If the character is not 0-9, it is ignored and not echoed
181  if((character>='0') && (character<='9')) {
182  number = 10*number+(character-'0'); // this line overflows if above 4294967295
183  length++;
184  UART_OutChar(character);
185  }
186 // If the input is a backspace, then the return number is
187 // changed and a backspace is outputted to the screen
188  else if((character==BS) && length){
189  number /= 10;
190  length--;
191  UART_OutChar(character);
192  }
193  character = UART_InChar();
194  }
195  return number;
196 }

213 //---------------------UART_InUHex----------------------------------------
214 // Accepts ASCII input in unsigned hexadecimal (base 16) format
215 // Input: none
216 // Output: 32-bit unsigned number
217 // No '$' or '0x' need be entered, just the 1 to 8 hex digits
218 // It will convert lower case a-f to uppercase A-F
219 // and converts to a 16 bit unsigned number
220 // value range is 0 to FFFFFFFF
221 // If you enter a number above FFFFFFFF, it will return an incorrect value
222 // Backspace will remove last digit typed
223 uint32_t UART_InUHex(void){
224 uint32_t number=0, digit, length=0;
225 char character;
226  character = UART_InChar();
227  while(character != CR){
228  digit = 0x10; // assume bad
229  if((character>='0') && (character<='9')){
230  digit = character-'0';
231  }
232  else if((character>='A') && (character<='F')){
233  digit = (character-'A')+0xA;
234  }
235  else if((character>='a') && (character<='f')){
236  digit = (character-'a')+0xA;
237  }
238 // If the character is not 0-9 or A-F, it is ignored and not echoed
239  if(digit <= 0xF){
240  number = number*0x10+digit;
241  length++;
242  UART_OutChar(character);
243  }
244 // Backspace outputted and return value changed if a backspace is inputted
245  else if((character==BS) && length){
246  number /= 0x10;
247  length--;
248  UART_OutChar(character);
249  }
250  character = UART_InChar();
251  }
252  return number;
253 }

277 //------------UART_InString------------
278 // Accepts ASCII characters from the serial port
279 // and adds them to a string until <enter> is typed
280 // or until max length of the string is reached.
281 // It echoes each character as it is inputted.
282 // If a backspace is inputted, the string is modified
283 // and the backspace is echoed
284 // terminates the string with a null character
285 // uses busy-waiting synchronization on RDRF
286 // Input: pointer to empty buffer, size of buffer
287 // Output: Null terminated string
288 // -- Modified by Agustinus Darmawan + Mingjie Qiu --
289 void UART_InString(char *bufPt, uint16_t max) {
290 int length=0;
291 char character;
292  character = UART_InChar();
293  while(character != CR){
294  if(character == BS){
295  if(length){
296  bufPt--;
297  length--;
298  UART_OutChar(BS);
299  }
300  }
301  else if(length < max){
302  *bufPt = character;
303  bufPt++;
304  length++;
305  UART_OutChar(character);
306  }
307  character = UART_InChar();
308  }
309  *bufPt = 0;
310 }
